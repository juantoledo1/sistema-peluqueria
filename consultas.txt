-- listar todos los turnos(id_turno, nombre de cliente, nombre de empleado, nombre del servicio, precio del servicio, hora y fecha)
SELECT 
    T.id_turno,
    C.nombre AS cliente,
    E.nombre AS empleado,
    S.nombre AS servicio,
    S.precio AS precio_servicio,
    T.fecha_hora
FROM 
    Turnos T
    JOIN Clientes C ON T.cliente_id = C.id_cliente
    JOIN Empleados E ON T.empleado_id = E.id_empleado
    JOIN Servicios S ON T.servicio_id = S.id_servicio;
