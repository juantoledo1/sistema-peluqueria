-- Ver lista de turnos con detalles de cliente, detalle y precio del servicio:
SELECT t.id_turno, t.fecha, t.hora, c.nombre AS nombre_cliente, c.apellido AS apellido_cliente, d.lugar AS detalle, s.nombre AS servicio, s.precio	
	FROM turnos t
	    inner join clientes c on t.id_cliente = c.id_cliente
        inner join detalles d on t.id_detalle = d.id_detalle
        inner join servicios s on t.id_servicio = s.id_servicio;
